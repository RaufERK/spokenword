2 . FFmpeg 7 — установка
bash
Копировать
Редактировать
cd /usr/local/bin
wget -O ffmpeg.tar.xz \
  https://johnvansickle.com/ffmpeg/releases/ffmpeg-release-amd64-static.tar.xz
FOLDER=$(tar tf ffmpeg.tar.xz | head -1 | cut -d/ -f1)
tar xf ffmpeg.tar.xz --strip-components=1 \
   "$FOLDER/ffmpeg" "$FOLDER/ffprobe"
chmod +x ffmpeg ffprobe
3 . Мини‑скрипт /usr/local/bin/start-hls.sh
bash
Копировать
Редактировать
#!/usr/bin/env bash
APP="$1"; NAME="$2"
LIVE_DIR=/srv/streaming/live
mkdir -p "$LIVE_DIR"

exec /usr/local/bin/ffmpeg -hide_banner -loglevel error \
 -i "rtmp://127.0.0.1/${APP}/${NAME}" \
 -c:v libx264 -preset veryfast -b:v 1500k -g 48 -keyint_min 48 -sc_threshold 0 \
 -c:a aac -ar 44100 -ac 2 -b:a 128k \
 -f hls -hls_time 4 -hls_list_size 0 \
 -hls_segment_filename "$LIVE_DIR/seg%03d.ts" \
 "$LIVE_DIR/playlist.m3u8"
Права: chmod 755 /usr/local/bin/start-hls.sh

4 . Фрагмент nginx.conf
nginx
Копировать
Редактировать
user  root;

rtmp {
  server {
    listen 1935;
    application live {
      live on;
      exec /usr/local/bin/start-hls.sh $app $name;
    }
  }
}

http {
  server {
    listen 80;
    server_name spoken-word.ru;

    location /live/ {
      alias /srv/streaming/live/;
      add_header Cache-Control no-cache;
      add_header Access-Control-Allow-Origin *;
      types { application/vnd.apple.mpegurl m3u8; video/mp2t ts; }
    }
  }
}
nginx -t && nginx -s reload

7 . Next.js: API‑Delete и цвета
ts
Копировать
Редактировать
// app/api/archive/[name]/route.ts
import { NextRequest, NextResponse } from 'next/server'
import fs from 'fs/promises'; import path from 'path'

export async function DELETE(
  _req: NextRequest, { params }: { params: { name: string } }
) {
  try {
    await fs.unlink(path.join('/srv/streaming/archive', params.name))
    return NextResponse.json({ success: true })
  } catch { return NextResponse.json({ success: false }, { status: 500 }) }
}
