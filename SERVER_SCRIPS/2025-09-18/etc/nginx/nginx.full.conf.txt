# 2) HTTPS: reverse proxy к приложению (порт 3013)
server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name amasters.info www.amasters.info;

    # сертификаты (уже выпущены)
    ssl_certificate     /etc/letsencrypt/live/amasters.info/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/amasters.info/privkey.pem;
    include             /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam         /etc/letsencrypt/ssl-dhparams.pem;

    # безопасные заголовки
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;
    add_header Permissions-Policy "geolocation=(), microphone=(), camera=()" always;
    # включи через 1–2 дня стабильной работы HTTPS:
    # add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;

    # общие прокси-хедеры
    proxy_http_version 1.1;
    proxy_set_header Host               $host;
    proxy_set_header X-Real-IP          $remote_addr;
    proxy_set_header X-Forwarded-For    $proxy_add_x_forwarded_for;
    proxy_set_header X-Forwarded-Proto  https;

    # основной роут → 127.0.0.1:3013
    location / {
        proxy_pass http://127.0.0.1:3013;

        # WebSocket / SSE
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";

        # таймауты и стриминг
        proxy_read_timeout 3600;
        proxy_send_timeout 3600;
        proxy_buffering off;
    }

    # статика Next.js (immutable)
    location ^~ /_next/static/ {
        proxy_pass http://127.0.0.1:3013;
        expires 1y;
        access_log off;
        add_header Cache-Control "public, max-age=31536000, immutable";
    }

    # Next/Image
    location ^~ /_next/image {
        proxy_pass http://127.0.0.1:3013;
        expires 30d;
        access_log off;
        add_header Cache-Control "public, max-age=2592000";
    }

    # медиа
    location ~* \.(?:png|jpg|jpeg|gif|webp|ico|svg|mp4|mp3|woff2?)$ {
        proxy_pass http://127.0.0.1:3013;
        expires 30d;
        access_log off;
        add_header Cache-Control "public, max-age=2592000";
    }

    # gzip
    gzip on;
    gzip_types text/css application/javascript application/json application/xml text/plain image/svg+xml;
    gzip_min_length 1024;

    # лимит запроса (под аплоады при необходимости)
    client_max_body_size 20m;

    # логи
    access_log /var/log/nginx/amasters_info_access.log;
    error_log  /var/log/nginx/amasters_info_error.log;
}

    listen [::]:443 ssl http2;
    server_name nutralogos.ru;

    ssl_certificate     /etc/letsencrypt/live/nutralogos.ru/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/nutralogos.ru/privkey.pem;
    include             /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam         /etc/letsencrypt/ssl-dhparams.pem;

    # Безопасные заголовки
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;
    add_header Permissions-Policy "geolocation=(), microphone=(), camera=()" always;
    # Включишь через 1–2 дня стабильного HTTPS:
    # add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;

    # Общие лимиты
    client_max_body_size 200M;
    client_body_timeout 60s;

    # ── Веб-приложение Next.js (порт 3000) ─────────────────────────
    location / {
        proxy_pass http://127.0.0.1:3000;
        proxy_http_version 1.1;

        proxy_set_header Host              $host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto https;

        proxy_set_header Upgrade           $http_upgrade;
        proxy_set_header Connection        "upgrade";

        proxy_read_timeout 300;
    }

    # Статика Next.js (immutable)
    location ^~ /_next/static/ {
        proxy_pass http://127.0.0.1:3000;
        expires 1y;
        access_log off;
        add_header Cache-Control "public, max-age=31536000, immutable";
    }

    # Next/Image
    location ^~ /_next/image {
        proxy_pass http://127.0.0.1:3000;
        expires 30d;
        access_log off;
        add_header Cache-Control "public, max-age=2592000";
    }

    # Прочая статика
    location ~* \.(?:png|jpg|jpeg|gif|webp|ico|svg|mp4|mp3|woff2?)$ {
        proxy_pass http://127.0.0.1:3000;
        expires 30d;
        access_log off;
        add_header Cache-Control "public, max-age=2592000";
    }

    # Загрузки (большие тела, без буферизации)
    location ^~ /api/upload {
        client_max_body_size 200M;
        proxy_pass http://127.0.0.1:3000;
        proxy_http_version 1.1;

        proxy_set_header Host              $host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto https;

        proxy_request_buffering off;
        proxy_buffering off;
        proxy_read_timeout 600;
        proxy_send_timeout 600;
    }

    # ── WebSocket (порт 3001) → wss://nutralogos.ru/ws/… ───────────
    location /ws/ {
        proxy_pass http://127.0.0.1:3001;
        proxy_http_version 1.1;

        proxy_set_header Upgrade           $http_upgrade;
        proxy_set_header Connection        "upgrade";
        proxy_set_header Host              $host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto https;

        proxy_read_timeout 300;
        proxy_send_timeout 300;
        proxy_buffering off;
    }

    # ── HTTP ручки WS (порт 3002) → https://nutralogos.ru/ws-http/… ─
    location /ws-http/ {
        proxy_pass http://127.0.0.1:3002;
        proxy_http_version 1.1;

        proxy_set_header Host              $host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto https;
    }

    # Сжатие
    gzip on;
    gzip_types text/css application/javascript application/json application/xml text/plain image/svg+xml;
    gzip_min_length 1024;

    # Логи
    access_log /var/log/nginx/nutralogos_ru_access.log;
    error_log  /var/log/nginx/nutralogos_ru_error.log;
}

server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name spoken-word.ru www.spoken-word.ru;

    ssl_certificate     /etc/letsencrypt/live/spoken-word.ru/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/spoken-word.ru/privkey.pem;
    include             /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam         /etc/letsencrypt/ssl-dhparams.pem;

    client_max_body_size 8G;

    # общий CORS (если нужен шире — оставь так)
    add_header Access-Control-Allow-Origin "*" always;

    # ─ HLS/видео ─
    location /live/ {
        alias /srv/streaming/live/;
        add_header Cache-Control "no-store, no-cache, must-revalidate, proxy-revalidate, max-age=0, s-maxage=0" always;
        add_header Pragma "no-cache" always;
        add_header Expires "0" always;
        add_header Surrogate-Control "no-store" always;
        add_header Access-Control-Allow-Origin "*";
        add_header Access-Control-Allow-Methods "GET, OPTIONS";
        add_header Access-Control-Allow-Headers "Range, Content-Type";
        add_header Access-Control-Expose-Headers "Content-Length, Content-Range";
        add_header Access-Control-Max-Age "86400";
        
        # Обработка OPTIONS запросов для CORS
        if ($request_method = OPTIONS) {
            add_header Access-Control-Allow-Origin "*";
            add_header Access-Control-Allow-Methods "GET, OPTIONS";
            add_header Access-Control-Allow-Headers "Range, Content-Type";
            add_header Access-Control-Max-Age "86400";
            add_header Content-Length 0;
            add_header Content-Type text/plain;
            return 204;
        }
        
        types { 
            application/vnd.apple.mpegurl m3u8;
            video/mp2t ts;
        }
        proxy_buffering off;
        etag off;
        if_modified_since off;
        expires -1;
    }

    location /archive/ {
        alias /srv/streaming/archive/;
        add_header Cache-Control "no-cache";
    }

    location /conf/ {
        alias /srv/streaming/conf/;
        add_header Cache-Control "no-cache";
        types { application/vnd.apple.mpegurl m3u8; video/mp2t ts; }
        proxy_buffering off;
    }

    location /conf-arch/ {
        alias /srv/streaming/conf-arch/;
        add_header Cache-Control "no-cache";
        add_header Access-Control-Allow-Origin "*";
    }

    # новый путь к conf-archive (как на новом сервере)
    location /conf-archive/ {
        alias /home/appuser/apps/spokenword/current/public/conf-archive/;
        add_header Cache-Control "no-cache";
        add_header Access-Control-Allow-Origin "*";
    }

	# тот же каталог под /watch-conf/
	location /watch-conf/ {
	    alias /home/appuser/apps/spokenword/current/public/conf-archive/;

	    # правильные MIME-типы для плеера в браузере
	    types {
	        video/mp4                       mp4 m4v;
	        video/webm                      webm;
	        video/quicktime                 mov;
	        application/vnd.apple.mpegurl   m3u8;
	        video/mp2t                      ts;
	        audio/mp4                       m4a;
	    }

	    # проигрывание "в браузере", а не скачивание
	    add_header Content-Disposition "inline" always;

	    # удобно для HLS/плееров
	    add_header Access-Control-Allow-Origin "*" always;
	    add_header Cache-Control "no-cache" always;

	    # для статики через alias проксирование не нужно
	    # proxy_buffering off;  # не требуется
	    gzip off;               # на всякий случай не сжимать видео
	}


    # ─ Прокси на Next.js (3005) ─
    location / {
        proxy_pass http://127.0.0.1:3005;
        proxy_http_version 1.1;

        proxy_set_header Host               $host;
        proxy_set_header X-Real-IP          $remote_addr;
        proxy_set_header X-Forwarded-For    $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto  https;

        proxy_set_header Upgrade            $http_upgrade;
        proxy_set_header Connection         "upgrade";

        proxy_read_timeout 600;
        proxy_send_timeout 600;
        proxy_buffering off;
    }

    gzip on;
    gzip_types text/css application/javascript application/json application/xml text/plain image/svg+xml;

    access_log /var/log/nginx/spoken_word_access.log;
    error_log  /var/log/nginx/spoken_word_error.log;
}
